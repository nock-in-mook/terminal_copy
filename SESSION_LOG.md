```python
def take_screenshot():
    ss_dir = get_screenshot_dir()
    ts = datetime.datetime.now().strftime('%Y%m%d_%H%M%S_%f')  # マイクロ秒追加で衝突防止
    path = os.path.join(ss_dir, f'ss_{ts}.png')
    log = os.path.join(ss_dir, '_screenshot.log')
    
    def _run():
        try:
            proc = subprocess.run(
                ['screencapture', '-i', path],
                capture_output=True, timeout=120
            )
            ok = os.path.exists(path)
            with open(log, 'a') as f:
                f.write(f'[{ts}] rc={proc.returncode} saved={ok} '
                        f'stderr={proc.stderr.decode()!r}\n')
        except Exception as e:
            with open(log, 'a') as f:
                f.write(f'[{ts}] error={e}\n')
    
    threading.Thread(target=_run, daemon=True).start()
```

- **マイクロ秒付きでファイル名衝突しない**（連打でも独立保存）
